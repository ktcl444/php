    /**
     * @param String $s
     * @return Integer
     */
    function lengthOfLongestSubstring($s) {
		if( !isset($s) || $s == '') return 0;
		$result= 0;
		$chars = [];
		$s_array = str_split($s);
		$length = count($s_array);
		for($i=0;$i<$length;$i++){
			$char = $s_array[$i];
			if(in_array($char,$chars)){
					$chars = removeChar($chars,$char);
		
				}
					
					$chars[] = $char;
				//print_r($chars);
			$result =	 count($chars) > $result ? count($chars):$result;
		}
		
		return $result;
	}
	
	function removeChar($chars,$char)
	{
		$result = [];
		$length = count($chars);
		$find = PHP_INT_MAX;
		for($i=0;$i<$length;$i++){
			if($chars[$i] == $char)
			{
				$find = $i;
				continue;
			}
			if($i>$find){
				$result[] = $chars[$i];
			}
		}
		
		return $result;
	}
